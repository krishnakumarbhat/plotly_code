#include <stdio.h>
#include <dlfcn.h>

typedef double (*add_func)(double, double);
typedef double (*subtract_func)(double, double);
typedef double (*multiply_func)(double, double);
typedef double (*divide_func)(double, double);

int main() {
    // Load the dynamic library
    void* lib_handle = dlopen("./libMathLibrary.so", RTLD_LAZY);
    if (!lib_handle) {
        fprintf(stderr, "Error: %s\n", dlerror());
        return 1;
    }

    // Get function pointers
    add_func add = (add_func)dlsym(lib_handle, "add");
    subtract_func subtract = (subtract_func)dlsym(lib_handle, "subtract");
    multiply_func multiply = (multiply_func)dlsym(lib_handle, "multiply");
    divide_func divide = (divide_func)dlsym(lib_handle, "divide");

    if (!add || !subtract || !multiply || !divide) {
        fprintf(stderr, "Error: %s\n", dlerror());
        dlclose(lib_handle);
        return 1;
    }

    double a = 10.0;
    double b = 5.0;

    printf("Addition: %.2f\n", add(a, b));
    printf("Subtraction: %.2f\n", subtract(a, b));
    printf("Multiplication: %.2f\n", multiply(a, b));
    printf("Division: %.2f\n", divide(a, b));

    // Close the library
    dlclose(lib_handle);

    return 0;
}

    // Create a mask with only the kth bit set
