import os
import sys

# Input path from Command line
path = sys.argv[1]

# Log selection from the given input path
logs = [i for i in os.listdir(path) if i.endswith('.mf4')]

# Writing logs to file to be used for RESIM as a input
with open(r'f_list.txt', 'w') as f:
    for i in logs:
        f.write('{}\{}\n'.format(path,i))
